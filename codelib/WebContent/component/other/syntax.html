使用解释

引用的JS和CSS 

			//shCore.js必须引入
			<script type="text/javascript"
				src="../../common/Syntax/scripts/shCore.js"></script>
			//shBrushJScript.js、shBrushJava.js、shBrushCss.js、shBrushXml.js等JS文件可以按需引入
			<script type="text/javascript"
					src="../../common/Syntax/scripts/shBrushJScript.js"></script>
			<script type="text/javascript"
					src="../../common/Syntax/scripts/shBrushJava.js"></script>
			<script type="text/javascript"
					src="../../common/Syntax/scripts/shBrushCss.js"></script>
			<script type="text/javascript"
					src="../../common/Syntax/scripts/shBrushXml.js"></script>
			//除了shCoreDefault.css外，还有多种css主题可选，在common/Syntax/styles目录下
			<link type="text/css" rel="stylesheet"
					href="../../common/Syntax/styles/shCoreDefault.css" />

页面<pre/>元素 

			//将代码包含于<pre></pre>
			//标签中，标签的class中定义brush属性
			//该属性的值可以为js\java\xml\css等
			//还可以为c\c++，不过需要额外的引入一些JS文件
			//这些JS包含在common/Syntax/scripts目录下
			<pre class="brush:java;">
		    	public class HelloWorld{
		    		public static void main(String[] args){
		    			System.out.println("Hello world!");
		    		}
		    	}
			</pre>
