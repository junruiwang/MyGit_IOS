使用(JavaScript代码) 

	
			Ext.onReady(function() {
				Ext.QuickTips.init();
				var sm = new Ext.grid.CheckboxSelectionModel({singleSelect:false});
				//定义后台JSON数据的阅读器
				//需要关注的是totalProperty和root两个属性
				//totalProperty表示总记录数
				//root对象表示具体的数据
				var reader = new Ext.data.JsonReader( //阅读器
					{totalProperty:'total',root : 'root'}, 
					[{
						name : 'id'
					},{
						name : 'title'
					},{
						name : 'content'
					},{
						name : 'beginDate.date'
					},{
						name : 'overDate.date'
					},{
						name : 'type'
					},{
						name : 'channelId'
					}]
				);
				//此处定义store
				var store=new Ext.data.Store({
					url : 'system/queryAfficheAjax.action',
					reader : reader
				});
				store.baseParams['sysAffiche.type'] = -999;
				var cm = new Ext.grid.ColumnModel([//表头
					{
						header : "标识",
						width : 20,
						sortable : true,
						dataIndex : 'id'
					},{
						header : "公告标题",
						width : 20,
						sortable : true,
						dataIndex : 'title'
					},{
						header : "公告内容",
						width : 100,
						sortable : true,
						dataIndex : 'content',
						css : 'overflow: hidden; text-overflow:ellipsis;',
						renderer : function(val) {
							return "" + val + "";
						}
					},{
						header : "生效开始日期",
						width : 20,
						sortable : true,
						dataIndex : 'beginDate.date'
					},{
						header : "生效结束日期",
						width : 20,
						sortable : true,
						dataIndex : 'overDate.date'
					},{
						header : "公告类型",
						width : 20,
						sortable : true,
						dataIndex : 'type',
						renderer : function(val) {
							return typeCN[val] != undefined ? typeCN[val] : val;
						}
					},{
						header : "受理渠道属地",
						width : 20,
						sortable : true,
						dataIndex : 'channelId',
						renderer : function(val) {
							return channelCn[val] != undefined ? channelCn[val] : "所有受理渠道属地";
						}
					}
				]);
			    var afficheTable = new Ext.grid.GridPanel({ //表格内容
					id : 'afficheTable',
					store : store,
					cm : cm,
					loadMask : true,
					trackMouseOver : false,
					viewConfig : {
						forceFit : true
					},
					bbar:new Ext.PagingToolbar({
						pageSize : 10,
						displayInfo : true,
						store:store,
						displayMsg : '显示第: {0} 条到 {1} 条记录，一共 {2}条',
						emptyMsg : '无数据'
					}),
					autoWidth:false,
					//autoHeight : true,
					height : 300,
					renderTo : 'sysAfficheEXT'
				});
				//afficheTable加载时绑定rowclick事件
				afficheTable.on('rowclick',function(grid, rowIndex, e){
					var store = grid.getStore();
					var row = store.data.items[rowIndex];
					//参数bein
					$('#afficheId').val(row.get("id"));
				}); 
				store.load({params:{limit:10,start:0}});
				cm.setHidden(0,true);
			});
